#/project.code-workspace 로 프로젝트 통합
{
    "folders": [
      { "path": "backend" },
      { "path": "frontend" }
    ],
    "settings": {
      "editor.formatOnSave": true,
      "files.exclude": {
        "**/node_modules": true,
        "**/target": true
      }
    }
  }

#/.vscode/tasks.json 작업실행
{
    "version": "2.0.0",
    "tasks": [
    {
    "label": "Start Backend",
    "type": "shell",
    "command": "mvnw.cmd spring-boot:run -Dspring-boot.run.arguments=--server.port=8080",
    "options": {
    "cwd": "${workspaceFolder}/backend",
    "shell": {"executable": "cmd.exe","args": ["/c"]}
    },
    "problemMatcher": []
    },
    {
    "label": "Start Frontend",
    "type": "shell",
    "command": "npm install && npm run dev",
    "options": {
    "cwd": "${workspaceFolder}/frontend",
    "shell": {"executable": "cmd.exe","args": ["/c"]}
    },
    "problemMatcher": []
    },
    {
    "label": "Start Both (Backend + Frontend)",
    "dependsOn": ["Start Backend", "Start Frontend"],
    "dependsOrder": "parallel"
    }
    ]
}
